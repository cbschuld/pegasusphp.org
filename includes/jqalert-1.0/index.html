jqalert Examples

View a regular alert example 
	$( function() {
		$("#alert").jqalert( { message: "This is just a normal alert message with nothing fancy in it other than just a message (like an alert() call)" } );
	} );
View a regular alert example with a title 
	$( function() {
		$("#alertt").jqalert( { title: "Just some Information", message: "This is just a normal alert message with nothing fancy in it other than just a message (like an alert() call)" } );
	} );
View a regular alert example with a fixed width 
	$( function() {
		$("#alertw").jqalert( { width: 450, message: "This is just a normal alert message with nothing fancy in it other than just a message (like an alert() call) but has a fixed with set" } );
	} );
View Error Example 
	$( function() {
		$("#error").jqalert( { error: true, message: "This is just a normal error message without any additional settings" } );
	} );
View Error Example with Title 
	$( function() {
		$("#errort").jqalert( { error: true, title: "Title Goes Here", message: "This is just a normal error message without any additional settings" } );
	} );
View Warning Example 
	$( function() {
		$("#warning").jqalert( { warning: true, message: "This is just a normal warning message" } );
	} );
View Information Example 
	$( function() {
		$("#info").jqalert( { info: true, message: "This is just a normal info message without any additional settings" } );
	} );
View Help Example 
	$( function() {
		$("#help").jqalert( { help: true, message: "This is just a normal help message without any additional settings" } );
	} );
View a simple Question Example 
	$( function() {
		$("#yesno").jqalert( {
								info: true,
								message: "Would you like to delete XYZ?<br/><em>(keep in mind there is no way to undo this operation<em>",
								success: function() { alert("you hit yes"); },
								failure: function() { alert("you hit no"); }
							} );
	} );
View a Full Question Example 
	$( function() {
		$("#yesno").jqalert( {
								info: true,
								title: "Would you like to delete XYZ?",
								message: "Click on the Yes button to delete XYZ!<br/><em>(keep in mind there is no way to undo this operation)<em>",
								success: function() { alert("you hit yes"); },
								failure: function() { alert("you hit no"); }
							} );
	} );
